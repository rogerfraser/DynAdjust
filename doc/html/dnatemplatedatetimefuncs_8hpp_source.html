79template <class T>
80void dateSINEXFormat(T* os, const boost::gregorian::date& theDate, bool calculateSeconds=false)
81{
82 // Time stamps must be in the following format: YY:DOY:SECOD
83 // where:
84 // YY : year
85 // DOY : day of year
86 // SECOD : sec of day (e.g. the epoch)
87 // Example:
88 // 95:120:86399 denotes April 30, 1995 (23:59:59UT).
89 //
90
91 std::stringstream year;
92 boost::posix_time::time_facet* facet(new boost::posix_time::time_facet("%y"));
93
94 year.imbue(std::locale(year.getloc(), facet));
95 boost::posix_time::ptime dateTime(theDate);
96 year << dateTime;
97
98 std::stringstream dayofyear;
99 dayofyear << std::right << std::setw(3) << theDate.day_of_year();
100 std::string dayofyearstr(dayofyear.str());
101 dayofyearstr = findandreplace<std::string>(dayofyearstr, " ", "0");
102
103 *os <<
104 std::right << std::setw(2) <<
105 year.str() << ":" << // year (YY)
106 std::right << std::setw(3) <<
107 dayofyearstr << ":"; // day of year
108
109 if (calculateSeconds)
110 {
111 boost::posix_time::ptime currentTime(boost::posix_time::second_clock::local_time());
112 boost::posix_time::time_duration timeInterval(currentTime - dateTime);
113
114 std::stringstream secondsofday;
115 secondsofday << std::left << std::setw(5) << timeInterval.total_seconds();
116 *os << std::left << std::setw(5) << secondsofday.str();
117 return;
118 }
119
120 *os << "00000";
121}

262template <class T>
263// calculate average and test for year cross over
264void year_doy_Average(const T& start_year, const T& end_year,
265 const T& start_doy, const T& end_doy,
266 T& average_year, T& average_doy)
267{
268 if (start_year != end_year)
269 {
270 average_doy = average(start_doy, start_doy + end_doy);
271 if (average_doy > DAYS_PER_YEAR)
272 {
273 average_year = end_year;
274 average_doy -= static_cast<T>(DAYS_PER_YEAR);
275 }
276 else
277 average_year = start_year;
278 return;
279 }
280
281 average_year = start_year;
282
283 if (start_doy != end_doy)
284 average_doy = average(start_doy, end_doy);
285 else
286 average_doy = start_doy;
287}

289template <class T>
290// epochFrom -> epochTo
291T yearFraction(const boost::gregorian::date& epoch)
292{
293 // Day of year, less 0.5 day correction to
294 // bring to middle of day, divided by number
295 // of days in a year
296 //return (dateDOY<T>(epoch) - 0.5) / DAYS_PER_YEAR;
297
298 short leap(0);
299 T days_in_year(365);
300
301 if (fmod((T)epoch.year(), 400.) == 0)
302 {
303 leap = 1;
304 days_in_year = 366;
305 }
306 else if (fmod((T)epoch.year(), 100.) == 0)
307 leap = 0;
308 else if (fmod((T)epoch.year(), 4.) == 0)
309 {
310 leap = 1;
311 days_in_year = 366;
312 }
313 else
314 leap = 0;
315
316 T day_of_year(CUMULATIVE_DAY_YEAR::cumulativeDays[leap][epoch.month() - 1] + epoch.day());
317
318 return (day_of_year - (T)0.5) / days_in_year;
319}

358template <class T>
359// dateString = dd.mm.yyyy
360boost::gregorian::date dateFromString_safe(const std::string& dateString)
361{
362 std::vector<std::string> vdateList;
363 SplitDelimitedString<std::string>(dateString, std::string("."), &vdateList);
364
365 try {
366 // boost date requires input in the following order: year, month, day
367 return boost::gregorian::date(
368 LongFromString<UINT32>(vdateList.at(2)), // year
369 LongFromString<UINT32>(vdateList.at(1)), // month
370 LongFromString<UINT32>(vdateList.at(0))); // day
371 }
372 catch (std::runtime_error& e)
373 {
374 std::stringstream ss;
375 ss << "dateFromString_safe(): Could not parse the date string \"" << dateString << "\"" << std::endl <<
376 " Details: " << e.what() << std::endl;
377 throw boost::enable_current_exception(std::runtime_error(ss.str()));
378 }
379 catch (...)
380 {
381 std::stringstream ss;
382 ss << "dateFromString_safe(): Could not parse the date string \"" << dateString << "\"." << std::endl <<
383 " Check that the epoch is formatted as dd.mm.yyyy" << std::endl;
384 throw boost::enable_current_exception(std::runtime_error(ss.str()));
385 }
386}

388template <class T, class S>
389// dateString = ddd yyyy
390T dateFromStringstream_doy_year(S& thedate)
391{
392 T parsedDate;
393 // From boost_1_54_0/doc/html/date_time/date_time_io.html
394 // %j : Day of year as decimal from 001 to 366 for leap years, 001 - 365 for non-leap years.
395 // "060" => Feb-29
396 // %y : Two digit year
397 // "05" => 2005
398 // %Y : Four digit year
399 // "2005" => 2005
400 // But! 94 => 2094!
401 // So use %Y"
402 thedate.imbue(std::locale(thedate.getloc(), new boost::gregorian::date_input_facet("%j %Y")));
403 thedate >> parsedDate;
404 return parsedDate;

428template <class T, class D>
429// date = yyyy.yyyyy
430T dateFromDouble_doy_year(const D& thedate)
431{
432 T parsedDate;
433 D year(floor(thedate));
434 D year_fraction(thedate - year);
435 D days_in_year(365.0);
436
437 if (fmod(year, 400.) == 0)
438 days_in_year += 1.0;
439 else if (fmod(year, 4.) == 0)
440 days_in_year += 1.0;
441
442 // Calculate integer day from floating point.
443 // If this rounds up or down, the net result is a day before
444 // or a day after. For some applications, this might cause
445 // one some concern. But for reference frame geodesy and plate
446 // tectonics, not a lot happens in a day so this is quite acceptable.
447 D doy = year_fraction * days_in_year;
448 size_t idoy = static_cast<size_t>(doy);
449 if (idoy < 1)
450 idoy = 1;
451
452 // From boost_1_54_0/doc/html/date_time/date_time_io.html
453 // %j : Day of year as decimal from 001 to 366 for leap years, 001 - 365 for non-leap years.
454 // "060" => Feb-29
455 // %y : Two digit year
456 // "05" => 2005
457 // %Y : Four digit year
458 // "2005" => 2005
459 // But! 94 => 2094!
460 // So use %Y"
461 std::stringstream strdate;
462 if (idoy < 10.)
463 strdate << "00";
464 else if (idoy < 100.)
465 strdate << "0";
466 strdate << idoy << " " << std::fixed << std::setprecision(0) << year;
467 strdate.imbue(std::locale(strdate.getloc(), new boost::gregorian::date_input_facet("%j %Y")));
468 strdate >> parsedDate;
469 return parsedDate;
470}

505template <class T>
506// dateString = dd.mm.yyyy
507T dateFromString(const std::string& dateString)
508{
509 // The following works well when dateString is "03.02.2013", but fails
510 // if dateString is "3.2.2013".
511 //
512 //istringstream ss(dateString);
513 //ss.imbue(std::locale(std::locale::classic(), new boost::gregorian::date_input_facet("%e.%m.%Y")));
514 //T date;
515 //ss >> date;
516 //return date;
517 //
518 // The Boost.Gregorian documentation specifies that:
519 // %d Day of the month as decimal 01 to 31
520 // %e # Like %d, the day of the month as a decimal number, but a leading zero is replaced by a space
521 // The top part of the documentation states the following warning:
522 // The flags marked with a hash sign (#) are implemented by system locale and are known to be missing on some platforms
523
524 // Is today's date required?
525 if (boost::iequals(dateString, "today"))
526 return dateFromString_safe<T>(stringFromDate<T>(T(boost::gregorian::day_clock::local_day())));
527
528 // Parse manually.
529 return dateFromString_safe<T>(dateString);
530
531}

546template <typename T>
547T formattedDateTimeString()
548{
549 std::stringstream datetime_ss, stream;
550 boost::posix_time::time_facet* p_time_output = new boost::posix_time::time_facet;
551 std::locale special_locale (std::locale(""), p_time_output);
552
553 // special_locale takes ownership of the p_time_output facet
554 datetime_ss.imbue (special_locale);
555 (*p_time_output).format("%d-%m-%Y %X");

562template <typename S>
563S formatedElapsedTime(boost::posix_time::milliseconds* elapsed_time, S app_message)
564{
565 std::ostringstream ss_time;
566 boost::posix_time::ptime pt(boost::posix_time::ptime(boost::gregorian::day_clock::local_day(), *elapsed_time));
567
568 if (*elapsed_time < boost::posix_time::seconds(3))
569 {
570 boost::posix_time::time_facet* facet(new boost::posix_time::time_facet("%s"));
571 ss_time.imbue(std::locale(ss_time.getloc(), facet));
572 ss_time.str("");
573 ss_time << pt << "s";
574 }
575 else if (*elapsed_time < boost::posix_time::seconds(61))
576 {
577 boost::posix_time::time_facet* facet(new boost::posix_time::time_facet("%S"));
578 ss_time.imbue(std::locale(ss_time.getloc(), facet));
579 ss_time.str("");
580 ss_time << pt << "s";
581 }
582 else
583 ss_time << boost::posix_time::seconds(static_cast<long>(elapsed_time->total_seconds()));
584
585 size_t pos = std::string::npos;
586 std::string time_message = ss_time.str();
587 while ((pos = time_message.find("0s")) != std::string::npos)
588 time_message = time_message.substr(0, pos) + "s";
589
590 time_message = app_message + time_message + ".";
591
592 if ((pos = time_message.find(" 00.")) != std::string::npos)
593 time_message = time_message.replace(pos, 4, " 0.");
594 if ((pos = time_message.find(" 0.s")) != std::string::npos)
595 time_message = time_message.replace(pos, 4, " 0s");
596
597 return time_message;
598}
